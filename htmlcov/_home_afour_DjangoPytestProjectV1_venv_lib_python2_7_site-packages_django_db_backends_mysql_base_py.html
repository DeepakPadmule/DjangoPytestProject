53def adapt_datetime_warn_on_aware_datetime(value, conv): 

54 # Remove this function and rely on the default adapter in Django 2.0. 

55 if settings.USE_TZ and timezone.is_aware(value): 

56 warnings.warn( 

57 "The MySQL database adapter received an aware datetime (%s), " 

58 "probably from cursor.execute(). Update your code to pass a " 

59 "naive datetime in the database connection's time zone (UTC by " 

60 "default).", RemovedInDjango20Warning) 

61 # This doesn't account for the database connection's timezone, 

62 # which isn't known. (That's why this adapter is deprecated.) 

63 value = value.astimezone(timezone.utc).replace(tzinfo=None) 

64 return Thing2Literal(value.strftime("%Y-%m-%d %H:%M:%S.%f"), conv)

85class CursorWrapper(object): 

86 """ 

87 A thin wrapper around MySQLdb's normal cursor class so that we can catch 

88 particular exception instances and reraise them with the right types. 

89 

90 Implemented as a wrapper, rather than a subclass, so that we aren't stuck 

91 to the particular underlying representation returned by Connection.cursor().

98 def execute(self, query, args=None): 

99 try: 

100 # args is None means no string interpolation 

101 return self.cursor.execute(query, args) 

102 except Database.OperationalError as e: 

103 # Map some error codes to IntegrityError, since they seem to be 

104 # misclassified and Django would prefer the more logical place. 

105 if e.args[0] in self.codes_for_integrityerror: 

106 six.reraise(utils.IntegrityError, utils.IntegrityError(*tuple(e.args)), sys.exc_info()[2]) 

107 raise

265 ))

267 # levels will be used, e.g. "set tx_isolation='repeatable-read'". 

268 isolation_level = isolation_level.replace(' ', '-')

270 kwargs.update(options)

279 def init_connection_state(self): 

280 assignments = []

282 # SQL_AUTO_IS_NULL controls whether an AUTO_INCREMENT column on

289 assignments.append("TX_ISOLATION = '%s'" % self.isolation_level)

291 if assignments: 

292 with self.cursor() as cursor: 

293 cursor.execute('SET ' + ', '.join(assignments))

329 def check_constraints(self, table_names=None): 

330 """ 

331 Checks each table name in `table_names` for rows with invalid foreign 

332 key references. This method is intended to be used in conjunction with 

333 `disable_constraint_checking()` and `enable_constraint_checking()`, to 

334 determine if rows with invalid references were entered while constraint 

335 checks were off. 

336 

337 Raises an IntegrityError on the first invalid foreign key reference 

338 encountered (if any) and provides detailed information about the 

339 invalid reference in the error message. 

340 

341 Backends can override this method if they can more directly apply 

342 constraint checking (e.g. via "SET CONSTRAINTS ALL IMMEDIATE") 

343 """ 

344 cursor = self.cursor() 

345 if table_names is None: 

346 table_names = self.introspection.table_names(cursor) 

347 for table_name in table_names: 

348 primary_key_column_name = self.introspection.get_primary_key_column(cursor, table_name) 

349 if not primary_key_column_name: 

350 continue 

351 key_columns = self.introspection.get_key_columns(cursor, table_name) 

352 for column_name, referenced_table_name, referenced_column_name in key_columns: 

353 cursor.execute( 

354 """ 

355 SELECT REFERRING.`%s`, REFERRING.`%s` FROM `%s` as REFERRING 

356 LEFT JOIN `%s` as REFERRED 

357 ON (REFERRING.`%s` = REFERRED.`%s`) 

358 WHERE REFERRING.`%s` IS NOT NULL AND REFERRED.`%s` IS NULL 

359 """ % ( 

360 primary_key_column_name, column_name, table_name, 

361 referenced_table_name, column_name, referenced_column_name, 

362 column_name, referenced_column_name, 

363 ) 

364 ) 

365 for bad_row in cursor.fetchall(): 

366 raise utils.IntegrityError( 

367 "The row in table '%s' with primary key '%s' has an invalid " 

368 "foreign key: %s.%s contains a value '%s' that does not have a corresponding value in %s.%s." 

369 % ( 

370 table_name, bad_row[0], table_name, column_name, 

371 bad_row[1], referenced_table_name, referenced_column_name, 

372 ) 

373 )

384 def mysql_version(self): 

385 with self.temporary_connection() as cursor: 

386 cursor.execute('SELECT VERSION()')
